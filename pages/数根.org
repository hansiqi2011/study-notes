#+BEGIN_SRC c
#include<stdio.h>
int main(void) {
    //一般思路：2345->2,3,4,5->14->1,4->5
    char str[1001];
    int i, sum;
    while (1) {
        gets(str);
        if (str[0] == '0')
            break;
        for (i = 0, sum = 0; str[i] != '\0'; i++)
            sum += str[i] - '0';
        printf("%d\n", (sum - 1) % 9 + 1);
    }
    return 0;
}
//递归法
int func1(int num) {
    if (num > 9) {
        int sum = 0;
        do sum += num % 10;
        while (num /= 10);
        if (sum < 10) return sum;
        else return func1(sum);
    }
    else return num;
}
//迭代法
int func2(int num) {
    while (num > 9) {
        int sum = 0;
        do sum += num % 10;
        while (num /= 10);
        num = sum;
    }
    return num;
}
//公式法
int func3(int num) {
    return (num - 1) % 9 + 1;
}
